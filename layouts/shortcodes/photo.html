{{- $src := .Get "src" -}} {{- $caption := .Get "caption" -}} {{- $thumb := .Get "thumb" -}} {{- $width := .Get "width" -}} {{- $height := .Get "height" -}} {{- $thumb_width := .Get "thumb_width" -}} {{- $thumb_height := .Get "thumb_height" -}} {{- if not $src -}} {{- errorf "Photo shortcode requires src parameter to be set in %q" .Path -}} {{- end -}} {{- $thumbnail := "" -}} {{- $width_attr := "" -}} {{- $height_attr := "" -}} {{- $overlay_style := "" -}} {{- if $width }}{{ $width = int $width }}{{ else }}{{ $width = 0 }}{{ end -}} {{- if $height }} {{ $height = int $height }} {{ else }} {{ $height = 0 }} {{ end -}} {{- $default_width := 1200 -}} {{- if .Site.Params.thumbnailWidth -}} {{- $default_width = int .Site.Params.thumbnailWidth -}} {{- end -}} {{- $default_height := 0 -}} {{- if .Site.Params.thumbnailHeight -}} {{- $default_height = int .Site.Params.thumbnailHeight -}} {{- end -}} {{- $jpeg_quality := 95 -}} {{- if .Site.Params.thumbnailQuality -}} {{- $jpeg_quality = int .Site.Params.thumbnailQuality -}} {{- end -}} {{- if $thumb -}} {{- if $width -}} {{- $width_attr = printf "width=\"%d\"" $width -}} {{- end -}} {{- if $height -}} {{- $height_attr = printf "height=\"%d\"" $height -}} {{- end -}} {{- $overlay_style = printf "style=\"width:%dpx; height: %dpx;\"" $width $height -}} {{- else -}} {{- $image := "" -}} {{- if or (hasPrefix $src "http://") (hasPrefix $src "https://") -}} {{- $image = resources.GetRemote $src -}} {{- else -}} {{- $image = resources.Get $src -}} {{- end -}} {{- if not $image -}} {{- errorf "Could not get image %s" $src -}} {{- end -}} {{- $source_width := $image.Width -}} {{- if not $source_width -}} {{- errorf "Image width is empty %q" $image.Width -}} {{- end -}} {{- $source_height := $image.Height -}} {{- if not $source_height -}} {{- errorf "Image height is empty %q" $image.Height -}} {{- end -}} {{- $aspect_ratio := div (float $source_width) (float $source_height) -}} {{- if gt $width 0 -}} {{- $height = int (math.Round (div $width $aspect_ratio)) -}} {{- else -}} {{- if gt $height 0 -}} {{- $width = int (math.Round (mul $height $aspect_ratio)) -}} {{- else -}} {{- if gt $default_width 0 -}} {{- $width = $default_width -}} {{- $height = int (math.Round (div $width $aspect_ratio)) -}} {{- else -}} {{- if gt $default_height 0 -}} {{- $height = $default_height -}} {{- $width = int (math.Round (mul $height $aspect_ratio)) -}} {{- else -}} {{- $width = $source_width -}} {{- $height = $source_height -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- $thumbnail = $image.Resize (printf "%dx%d q%d" $width $height $jpeg_quality) -}} {{- $thumb = $thumbnail.Permalink -}} {{- $width_attr = printf "width=\"%d\"" $thumbnail.Width -}} {{- $height_attr = printf "height=\"%d\"" $thumbnail.Height -}} {{- $overlay_style = printf "style=\"width:%dpx; height: %dpx;\"" $thumbnail.Width $thumbnail.Height -}} {{- end -}}
